cmake_minimum_required(VERSION 3.28.0)
project(2d_cod_zombie VERSION 0.1.0 LANGUAGES C CXX)

include(CTest)
enable_testing()

add_executable(2d_cod_zombie WIN32 source/main.cpp source/map.cpp source/input.cpp source/classes.cpp source/mouse.cpp source/rounds.cpp source/niga.cpp source/menu.cpp source/hud.cpp)

add_executable(my_test source/test.cpp)
add_test(NAME MyTest COMMAND my_test)

target_link_libraries(2d_cod_zombie SDL2main SDL2 SDL2_image SDL2_ttf)
set(CPACK_PROJECT_NAME ${PROJECT_NAME})
set(CPACK_PROJECT_VERSION ${PROJECT_VERSION})
include(CPack)
