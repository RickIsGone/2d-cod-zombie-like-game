cmake_minimum_required(VERSION 3.28.0)
project(2d_cod_zombie VERSION 0.1.0 LANGUAGES C CXX)

include(CTest)
enable_testing()

add_executable(2d_cod_zombie  source/main.cpp source/map.cpp source/input.cpp source/classes.cpp source/mouse.cpp source/rounds.cpp source/niga.cpp source/menu.cpp source/hud.cpp source/consolle.cpp)

add_executable(my_test source/test.cpp)
add_test(NAME MyTest COMMAND my_test)

set(SDL2_DIR "${CMAKE_CURRENT_SOURCE_DIR}/src/lib/cmake/SDL2")
set(SDL2_IMAGE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/src/lib/cmake/SDL2_image")
set(SDL2_TTF_DIR "${CMAKE_CURRENT_SOURCE_DIR}/src/lib/cmake/SDL2_ttf")
set(SDL2_MIXER_DIR "${CMAKE_CURRENT_SOURCE_DIR}/src/lib/cmake/SDL2_mixer")

find_package(SDL2 REQUIRED)
find_package(SDL2_image REQUIRED)
find_package(SDL2_ttf REQUIRED)
find_package(SDL2_mixer REQUIRED)

include_directories(${SDL2_INCLUDE_DIRS} ${SDL2_IMAGE_INCLUDE_DIRS} ${SDL2_TTF_INCLUDE_DIRS} ${SDL2_MIXER_INCLUDE_DIRS})

target_link_libraries(2d_cod_zombie "${CMAKE_CURRENT_SOURCE_DIR}/src/lib/libSDL2.a" "${CMAKE_CURRENT_SOURCE_DIR}/src/lib/libSDL2_image.a" "${CMAKE_CURRENT_SOURCE_DIR}/src/lib/libSDL2_ttf.a" "${CMAKE_CURRENT_SOURCE_DIR}/src/lib/libSDL2_mixer.a" Setupapi Imm32 Cfgmgr32 Winmm Version Rpcrt4)
set(CPACK_PROJECT_NAME ${PROJECT_NAME})
set(CPACK_PROJECT_VERSION ${PROJECT_VERSION})
include(CPack)
